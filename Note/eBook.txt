    var index = 0
    fun changeImg(view: View) {
        var btn = view as Button
        when (btn.id) {
            R.id.next_btn -> {
                if(index == img.size-1) {
                    index = 0
                }
                else
                    index++
                poke.setImageResource(img[index])

            }
            R.id.pre_btn -> {
                if(index == 0) {
                    index = img.size - 1
                }
                else
                    index--
                poke.setImageResource(img[index])

            }
            R.id.jump_btn -> {
                var indexPage = index_page.text.toString().toInt()
                poke.setImageResource(img[indexPage-1])

            }
        }
    }
